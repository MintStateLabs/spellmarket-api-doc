The Product resource lets you manage products in a sales channel.
You can add or update product Media. Products can be organized by grouping them into a Collection.

```graphql
type Product implements Node, Publishable {
  availablePublicationCount: Int!
  item: SellableItem!
  createdAt: DateTime!
  customProductType: String
  defaultCursor: String!
  description(
  truncateAt: Int
): String!
  descriptionHtml: HTML!
  featuredImage: Image
  featuredMedia: Media
  handle: String
  id: ID!
  mediaCount: Int!
  onlineStorePreviewUrl: URL
  onlineStoreUrl: URL
  options(
  first: Int
): Int!
  productType: String!
  publicationCount(
  onlyPublished: Boolean = true
): Int!
  publishedAt: DateTime!
  publishedOnCurrentPublication: Boolean!
  publishedOnPublication(
  publicationId: ID!
): Boolean!
  resourcePublicationOnCurrentPublication: ResourcePublication
  seo: SEO!
  status: ProductStatus!
  tags: StringConnection!
  title: String!
  totalInventory: Int!
  tracksInventory: Boolean!
  updatedAt: DateTime!
  vendor: String!
  productCollections(
  first: Int
  after: String
  last: Int
  before: String
  reverse: Boolean = false
  sortKey: CollectionSortKeys
  query: String
): ProductCollectionConnection!
  images(
  first: Int
  after: String
  last: Int
  before: String
  reverse: Boolean = false
  sortKey: ProductImageSortKeys = POSITION
): ImageConnection!
  media(
  first: Int
  after: String
  last: Int
  before: String
  reverse: Boolean = false
  sortKey: ProductMediaSortKeys = POSITION
): MediaConnection!
  resourcePublications(
  onlyPublished: Boolean = true
  first: Int
  after: String
  last: Int
  before: String
  reverse: Boolean = false
): ResourcePublicationConnection!
  unpublishedPublications(
  first: Int
  after: String
  last: Int
  before: String
  reverse: Boolean = false
): PublicationConnection!
}
```
