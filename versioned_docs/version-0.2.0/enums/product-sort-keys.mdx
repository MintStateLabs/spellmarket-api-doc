```graphql
enum ProductSortKeys {
  TITLE
  PRODUCT_TYPE
  PROJECT
  INVENTORY_TOTAL
  UPDATED_AT
  CREATED_AT
  PUBLISHED_AT
  ID
  RELEVANCE
}
```
